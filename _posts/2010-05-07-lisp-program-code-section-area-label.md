本段代码是 [AutoCAD工程测量工具集](https://www.cehui.ren/posts/acad-toolkit/) 中的一个功能的源代码，可以单独运行。

代码中各命令功能如下：

- **zzA**与**zzArea**：标注面积到区域中心
- **zzArea2Table**：输出面积到AutoCAD表格
- **zzArea2File**：输出面积到文件

(defun c:zzArea2File nil (AreaLabel 2))

    (vla-put-textalignmentpoint text (vlax-3D-point point))

	      )

	(setq result

		     (vl-catch-all-apply
		       'vlax-invoke
		       (list Shell 'Open target)

		     )
		   )
		 )

	(vlax-release-object Shell)
	)
      )
    result

  (defun _Select  (msg pred func init / e)
    (setq pred (eval pred))
    (while
      (progn (setvar 'ERRNO 0)
	     (apply 'initget init)
	     (setq e (func msg))
	     (cond
	       ((= 7 (getvar 'ERRNO))
		(princ "\nMissed, try again.")
		)
	       ((eq 'STR (type e))
		nil
		)
	       ((vl-consp e)
		(if (and pred (not (pred (setq e (car e)))))
		  (princ "\nInvalid Object Selected.")
		  )
		)
	       )
	     )
      )
    e
    )
